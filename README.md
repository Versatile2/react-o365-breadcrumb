```bash
npm install --save react-versatile2-breadcrumb
```

## Usage

```tsx
import * as React from 'react'

import MyComponent from 'react-versatile2-breadcrumb'

class Example extends React.Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
